DSP Lab 6 — DFT Computation

clc;
clear all;
close all;

% DFT computation
x = input('enter the input sequence');
N = input('enter the value of N point DFT');

n = [0:1:N-1]

ylabel('phase');

DSP Lab 8 — IDFT using Formula

clc;
clear all;
close all;

% Input frequency domain sequence
Xk = input('Enter frequency domain sequence X(k): ');

N = length(Xk);

% Index
n = 0:N-1;
k = 0:N-1;

% IDFT computation using formula
xn = (1/N) * (Xk * exp(j*2*pi/N * (k'*n)));

% Display result
disp('Time domain sequence x(n) is:');
disp(xn);

% Plot original sequence
subplot(2,1,1);
stem(k, abs(Xk));
title('Input Magnitude Spectrum X(k)');
xlabel('k');
ylabel('|X(k)|');

% Plot IDFT result
subplot(2,1,2);
stem(n, real(xn));
title('Recovered Time Domain Signal x(n)');
xlabel('n');
ylabel('Amplitude');

DSP Lab 9 — Low Pass Butterworth Filter

clear all;

Rp = 4;   % Pass band attenuation in dB
Rs = 30;  % Stop band attenuation in dB

fp = 400; % pass band frequency in Hz
fs = 800; % stop band frequency in Hz
F = 2000; % Sampling frequency in Hz

omp = 2*fp/F;
oms = 2*fs/F;

% To find cutoff frequency and order of the filter
[n,wc] = buttord(omp,oms,Rp,Rs)

% system function of the filter
[b,a] = butter(n,wc)

w = 0:.01:pi;

[h,om] = freqz(b,a,w,'whole');

m = 20*log10(abs(h));
an = angle(h);

subplot(2,1,1);
plot(om/pi,m);
grid;

ylabel('Gain in dB');
xlabel('Normalised frequency');
title('magnitude response');

subplot(2,1,2);
plot(om/pi,an);
grid;

ylabel('Phase in radians');
xlabel('Normalised frequency');
title('phase response');
